27 * Create a basic button on the specified pin.

29 * @param pin The pin connected to the button.

31 TBasicButton(uint8_t pin)

33
34 /**
35 * Initialise the button. Called from setup() of an Arduino program.

38 button_.begin();
39 }
40
41 /**
42 * Determine if the button is currently pressed.
43 *
44 * @return True if pressed, else false.
